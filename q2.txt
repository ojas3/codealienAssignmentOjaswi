#include <iostream>
using namespace std;

int main() {
	int st[]={2,3,4,6,7,8,9,10,13,15,18,19,23};
	int len=sizeof(st)/sizeof(st[0]);
	int sum =0;
	int last =0;
	for (int i=0; i<len;i++)
	{
	    if(st[i]%3==0)
	    {
	    sum+=st[i];
	        last=st[i];
	    }
	}
	cout<<"sum="<<sum<<", last="<<last;
	
	return 0;
}
